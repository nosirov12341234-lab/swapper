export function ConnectButton() {
  const [tonConnectUI] = useTonConnectUI();
  const [buttonText, setButtonText] = useState('Connect Wallet');

  useEffect(() => {
    if (tonConnectUI.connected) {
      setButtonText('Wallet Connected');
    } else {
      setButtonText('Connect Wallet');
    }
  }, [tonConnectUI.connected]);

  const handleConnectClick = () => {
    if (tonConnectUI.connected) {
      tonConnectUI.disconnect();
    } else {
      tonConnectUI.openModal();
    }
  };

    <button
      onClick={handleConnectClick}
      className={`
        px-6 py-3 rounded-xl font-semibold transition-all duration-300
        ${tonConnectUI.connected ? 'bg-tonGreen hover:bg-green-700' : 'bg-tonBlue hover:bg-blue-700'}
        text-white shadow-lg transform hover:scale-105
      `}
    >
      {buttonText}
    </button>
  );
}

  return coreToNano(amount.toString());
};

export const fromNano = (amount: BN | bigint | string | number): string => {
  if (typeof amount === 'number') {
    amount = amount.toString();
  }
  return coreFromNano(amount);

  const getSwapRate = (from: Token, to: Token) => {

    if (from.symbol === 'JT' && to.symbol === 'TON') return 2;   // 1 JT = 2 TON
    return 1; // Fallback

      const rate = getSwapRate(fromToken, toToken);
